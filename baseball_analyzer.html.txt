<script>
let selectedMotion = 'pitching';
let uploadedFile = null;
let uploadedBase64 = null;
let currentResult = null;

const motionLabels = {
  pitching: 'ピッチング（投球）',
  batting: 'バッティング（打撃）',
  fielding: '守備・スローイング',
  running: '走塁・身体操作'
};

function selectMotion(el) {
  document.querySelectorAll('.motion-btn').forEach(b => b.classList.remove('active'));
  el.classList.add('active');
  selectedMotion = el.dataset.motion;
  updateMotionTag();
}

function updateMotionTag() {
  const tag = document.getElementById('motion-tag-display');
  tag.textContent = '選択中：' + motionLabels[selectedMotion];
  tag.classList.add('show');
}

function handleDragOver(e) {
  e.preventDefault();
  document.getElementById('upload-area').classList.add('dragover');
}

function handleDragLeave(e) {
  document.getElementById('upload-area').classList.remove('dragover');
}

function handleDrop(e) {
  e.preventDefault();
  document.getElementById('upload-area').classList.remove('dragover');
  const file = e.dataTransfer.files[0];
  if (file) processFile(file);
}

function handleFile(e) {
  const file = e.target.files[0];
  if (file) processFile(file);
}

function processFile(file) {
  uploadedFile = file;
  const preview = document.getElementById('preview-container');
  const nameDisplay = document.getElementById('file-name-display');
  preview.innerHTML = '';
  preview.style.display = 'block';
  nameDisplay.style.display = 'block';
  nameDisplay.textContent = '📎 ' + file.name + ' (' + (file.size / 1024 / 1024).toFixed(1) + 'MB)';

  const reader = new FileReader();
  reader.onload = function(e) {
    uploadedBase64 = e.target.result.split(',')[1];
    if (file.type.startsWith('image/')) {
      const img = document.createElement('img');
      img.src = e.target.result;
      preview.appendChild(img);
    } else if (file.type.startsWith('video/')) {
      const video = document.createElement('video');
      video.src = e.target.result;
      video.controls = true;
      video.style.maxWidth = '100%';
      preview.appendChild(video);
    }
  };
  reader.readAsDataURL(file);
}

async function runAnalysis() {
  const name = document.getElementById('player-name').value || '選手';
  const grade = document.getElementById('player-grade').options[document.getElementById('player-grade').selectedIndex].text;
  const hand = document.getElementById('player-hand').value === 'right' ? '右利き' : '左利き';
  const note = document.getElementById('player-note').value;
  const motionLabel = motionLabels[selectedMotion];

  updateMotionTag();

  document.getElementById('loading').style.display = 'block';
  document.getElementById('result-section').style.display = 'none';
  document.getElementById('analyze-btn').disabled = true;

  const loadingTexts = [
    'バイオメカニクスデータを解析中...',
    '関節角度・重心移動を確認中...',
    'ドリルプランを作成中...',
    'レポートを生成中...'
  ];
  let li = 0;
  const ltimer = setInterval(() => {
    li = (li + 1) % loadingTexts.length;
    document.getElementById('loading-text').textContent = loadingTexts[li];
  }, 1800);

  const systemPrompt = `あなたは最先端のスポーツバイオメカニクスの専門家であり、少年野球コーチングのエキスパートです。
小学生の野球動作を分析し、最新のバイオメカニクス知識に基づいて指導レポートを提供します。
必ずJSON形式のみで回答してください。余計なテキストは一切出力しないでください。
出力するJSONの構造：
{
  "score": 数値(0-100),
  "title": "一言評価（例：バランスの良い投球フォーム）",
  "summary": "2〜3文の総合評価",
  "goodPoints": ["良好ポイント1", "良好ポイント2", "良好ポイント3"],
  "improvePoints": ["改善ポイント1", "改善ポイント2", "改善ポイント3"],
  "drills": [
    {
      "name": "ドリル名",
      "tag": "カテゴリ（例：体幹強化、柔軟性など）",
      "description": "ドリルの説明（2〜3文）",
      "points": "実施ポイント（3項目）",
      "youtubeQuery": "YouTubeで検索するキーワード（日本語）"
    }
  ],
  "coachAdvice": "コーチへのアドバイス（2〜3文）"
}`;

  const userContent = [];

  if (uploadedBase64 && uploadedFile && uploadedFile.type.startsWith('image/')) {
    userContent.push({
      type: 'image',
      source: {
        type: 'base64',
        media_type: uploadedFile.type,
        data: uploadedBase64
      }
    });
  }

  userContent.push({
    type: 'text',
    text: `選手情報：
- 名前：${name}
- 学年：${grade}
- 利き手：${hand}
- 分析対象動作：${motionLabel}
- 特記事項：${note || 'なし'}

${uploadedBase64 && uploadedFile.type.startsWith('image/') ? '上記の画像を分析してください。' : ''}
${uploadedFile && uploadedFile.type.startsWith('video/') ? '動画ファイルがアップロードされましたが、静止画像のみ分析可能です。動作の特記事項を参考に分析を行ってください。' : ''}
${!uploadedFile ? '画像が添付されていないため、特記事項と選手情報を元に一般的な分析を行ってください。' : ''}

最新のバイオメカニクス理論に基づき、小学生に適切な指導レポートをJSONで出力してください。`
  });

  try {
    const response = await fetch('https://api.anthropic.com/v1/messages', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        model: 'claude-sonnet-4-20250514',
        max_tokens: 1000,
        system: systemPrompt,
        messages: [{ role: 'user', content: userContent }]
      })
    });

    const data = await response.json();
    const rawText = data.content.map(b => b.text || '').join('');
    const clean = rawText.replace(/```json|```/g, '').trim();
    const result = JSON.parse(clean);
    currentResult = { result, name, grade, motionLabel };

    clearInterval(ltimer);
    displayResult(result, name, motionLabel);

  } catch (err) {
    clearInterval(ltimer);
    alert('分析中にエラーが発生しました。再度お試しください。\n' + err.message);
    document.getElementById('loading').style.display = 'none';
    document.getElementById('analyze-btn').disabled = false;
  }
}

function displayResult(r, name, motion) {
  document.getElementById('loading').style.display = 'none';
  document.getElementById('result-section').style.display = 'block';
  document.getElementById('analyze-btn').disabled = false;

  document.getElementById('result-score').textContent = r.score;
  document.getElementById('result-title').textContent = name + ' さん｜' + r.title;
  document.getElementById('result-summary').textContent = r.summary;

  const goodUl = document.getElementById('good-points');
  goodUl.innerHTML = '';
  (r.goodPoints || []).forEach(p => {
    const li = document.createElement('li');
    li.textContent = p;
    goodUl.appendChild(li);
  });

  const impUl = document.getElementById('improve-points');
  impUl.innerHTML = '';
  (r.improvePoints || []).forEach(p => {
    const li = document.createElement('li');
    li.textContent = p;
    impUl.appendChild(li);
  });

  const drillContainer = document.getElementById('drills-container');
  drillContainer.innerHTML = '';
  (r.drills || []).forEach((d, i) => {
    const card = document.createElement('div');
    card.className = 'drill-card';
    const ytUrl = 'https://www.youtube.com/results?search_query=' + encodeURIComponent(d.youtubeQuery + ' 少年野球');
    card.innerHTML = `
      <div class="drill-header">
        <div class="drill-name">ドリル${i + 1}｜${d.name}</div>
        <div class="drill-tag">${d.tag}</div>
      </div>
      <div class="drill-desc">${d.description}</div>
      <div class="drill-points"><p><strong>実施ポイント：</strong>${d.points}</p></div>
      <a class="youtube-link" href="${ytUrl}" target="_blank">▶ YouTubeで検索：${d.youtubeQuery}</a>
    `;
    drillContainer.appendChild(card);
  });

  document.getElementById('coach-advice').textContent = r.coachAdvice;
  document.getElementById('result-section').scrollIntoView({ behavior: 'smooth' });
}
